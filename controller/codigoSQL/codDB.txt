CREATE DATABASE SMILIPS;

USE SMILIPS;

CREATE TABLE usuario(usuarioID INT AUTO_INCREMENT PRIMARY KEY, nomeUsuario VARCHAR(100) NOT NULL, cpf_cnpj VARCHAR(19) UNIQUE NOT NULL, emailUsuario VARCHAR(100) UNIQUE NOT NULL, senhaUsuario VARCHAR(35) NOT NULL, endereco VARCHAR(45) NOT NULL, bairro VARCHAR(45) NOT NULL, complemento VARCHAR(45), telefone VARCHAR(20) NOT NULL, ftPerfil 	LONGBLOB);
